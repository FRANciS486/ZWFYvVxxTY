	}

		CLR_K8S_PATH="${env.WORKSPACE}/clr-k8s-examples"

		}
		stage('CNI') {
				dir(path: "$CLR_K8S_PATH") {
					sh './create_stack.sh cni'
		stage('Reset Stack') {
			}
		}
	}
		always {
			sh 'uname -a'
			sh 'swupd info'
